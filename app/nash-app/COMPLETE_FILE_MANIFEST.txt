# In your main dashboard file:

import gradio as gr
from nashville_sim_integration import create_nashville_simulation_tab

def create_app():
    with gr.Blocks(title="Nashville Incentive Dashboard") as app:
        gr.Markdown("# Dashboard")
        
        with gr.Tabs():
            # Existing tabs
            with gr.TabItem("Incentive Analytics"):
                create_incentive_tab()
            
            # NEW TAB
            with gr.TabItem("Nashville Simulation"):
                create_nashville_simulation_tab()
        
        return app

if __name__ == "__main__":
    demo = create_app()
    demo.launch()

# Then run:

# Open: http://localhost:7860

================================================================================
TROUBLESHOOTING
================================================================================

Problem: ModuleNotFoundError: No module named 'nashville_sim_data'
Solution: Ensure all .py files are in same directory as main dashboard

Problem: Blank visualizations
Solution: pip install matplotlib --upgrade

Problem: Dashboard slow to load
Solution: Enable caching with @lru_cache() decorator

Problem: Data looks incorrect
Solution: Check nashville_sim_data.py contains expected values
